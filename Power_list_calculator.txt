def exponents(bases, powers):
  results = []
  
  for base in bases:
    for exponent in powers:
      result = base ** exponent
      results.append(result)
      
  return(results)

#Uncomment the line below when your function is done
print(exponents([2, 3, 4], [1, 2, 3]))
