#First function (From fahrenheit to celsius)

def f_to_c(f_temp):
  c_temp = (f_temp - 32) * 5/9
  return c_temp

#Second function (From celsius to fahrenheit)

def c_to_f(c_temp):
  f_temp = c_temp * (9/5) +32
  return f_temp
